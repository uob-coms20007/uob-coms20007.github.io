$$
  \begin{array}{rcl}
    B &\to& F\\
      &\to& L \andop F\\
      &\to& L \andop L\\
      &\to& L \andop (B)\\
      &\to& L \andop (F \orop B)\\
      &\to& L \andop (L \orop B)\\
      &\to& L \andop (L \orop F)\\
      &\to& L \andop (L \orop L)\\
      &\to& \ff \andop (L \orop L)\\
      &\to& \ff \andop (\tt \orop L)\\
      &\to& \ff \andop (\tt \orop \ff)
  \end{array}
$$
